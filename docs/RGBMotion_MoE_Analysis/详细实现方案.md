# RGB + Motion MoE 详细实现方案

## 一、架构设计总览

### 1.1 网络整体架构

```
输入: RGB (H, W, 3) + Motion (H, W, 3) → 融合 → (H, W, 6)
                         ↓
                  ┌─────┴─────┐
                  ↓           ↓
           ┌──────────┐  ┌──────────┐
           │RGB分支   │  │Motion分支 │
           │(Conv×N) │  │(Conv×N)  │
           └────┬────┘  └────┬────┘
                ↓           ↓
           ┌────────────────────┐
           │   模态感知MoE层     │
           │ (RGB+M跨模态路由)   │
           └────────────────────┘
                         ↓
                  ┌─────┴─────┐
                  ↓           ↓
           ┌──────────┐  ┌──────────┐
           │深层特征  │  │检测Head  │
           │(C3k2×N) │  │ (Detect) │
           └──────────┘  └──────────┘
```

### 1.2 通道配置方案

| 方案 | RGB通道 | Motion通道 | 总通道 | 说明 |
|------|--------|-----------|--------|------|
| 方案A | 3 | 3 | 6 | 运动图伪彩色(推荐) |
| 方案B | 3 | 1 | 4 | 运动图灰度，节省通道 |
| 方案C | 3 | 3 | 6 | 分支独立处理后融合 |

## 二、运动信息提取模块

### 2.1 HomoMotion 集成方式

```python
# ultralytics/data/motion_extractor.py
import sys
sys.path.append('3rdparty/HomoMotion')

from homomotion import MotionDetector
import numpy as np
import cv2
from collections import deque

class HomoMotionExtractor:
    """HomoMotion 运动信息提取器"""

    def __init__(
        self,
        num_frames: int = 3,
        frame_interval: int = 1,
        feature_type: str = 'ORB',  # 可选: 'ORB', 'SIFT', 'CUDA_SIFT'
        nfeatures: int = 2000,
        diff_mode: str = 'mean',
        output_channels: int = 3,
        debug: bool = False
    ):
        """
        Args:
            num_frames: 用于计算运动的帧数
            frame_interval: 帧间隔
            feature_type: 特征提取算法
            nfeatures: 特征点数量
            diff_mode: 帧差计算模式 ('mean' 或 'subtract')
            output_channels: 输出通道数 (1 或 3)
            debug: 是否开启调试模式
        """
        self.num_frames = num_frames
        self.frame_interval = frame_interval
        self.output_channels = output_channels
        self.debug = debug

        # 初始化运动检测器
        self.detector = MotionDetector(
            feature_type=feature_type,
            num_frames=num_frames,
            frame_interval=frame_interval,
            diff_mode=diff_mode,
            nfeatures=nfeatures,
            debug=debug
        )

        # 帧缓冲
        self.frame_buffer = deque(maxlen=num_frames)

    def extract(self, current_frame: np.ndarray) -> np.ndarray:
        """
        提取运动信息图

        Args:
            current_frame: BGR 格式当前帧

        Returns:
            motion_map: 运动图 (单通道或3通道)
        """
        # 确保 BGR 格式
        if len(current_frame.shape) == 2:
            current_frame = cv2.cvtColor(current_frame, cv2.GRAY2BGR)

        self.frame_buffer.append(current_frame)

        # 帧数不足时返回空白图
        if len(self.frame_buffer) < self.num_frames:
            h, w = current_frame.shape[:2]
            if self.output_channels == 1:
                return np.zeros((h, w), dtype=np.uint8)
            else:
                return np.zeros((h, w, 3), dtype=np.uint8)

        # 计算运动图
        result = self.detector.process(list(self.frame_buffer))
        motion_map = result.motion_map

        # 归一化
        if motion_map.max() > motion_map.min():
            motion_map = ((motion_map - motion_map.min()) /
                         (motion_map.max() - motion_map.min()) * 255).astype(np.uint8)

        # 转为目标通道数
        if self.output_channels == 1:
            return motion_map.astype(np.uint8)
        else:
            # 转为伪彩色 (便于网络学习)
            return cv2.applyColorMap(motion_map, cv2.COLORMAP_JET)

    def reset(self):
        """重置帧缓冲"""
        self.frame_buffer.clear()
```

### 2.2 数据集类扩展

```python
# ultralytics/data/dataset_rgbmotion.py
from ultralytics.data import BaseDataset
from .motion_extractor import HomoMotionExtractor

class RGBMotionDataset(BaseDataset):
    """
    RGB + Motion 多模态数据集

    支持的数据目录结构:
    dataset/
    ├── visible/        # RGB 可见光图像
    │   ├── train/
    │   └── val/
    └── motion/        # 可选: 预计算的运动图
        ├── train/
        └── val/
    """

    def __init__(
        self,
        *args,
        motion_mode: str = 'extract',  # 'extract': 实时提取, 'load': 加载预计算
        motion_config: dict = None,
        **kwargs
    ):
        super().__init__(*args, **kwargs)

        self.motion_mode = motion_mode

        if motion_config is None:
            motion_config = {
                'num_frames': 3,
                'feature_type': 'ORB',  # ORB 更快
                'output_channels': 3,
            }

        self.motion_config = motion_config

        if motion_mode == 'extract':
            self.motion_extractor = HomoMotionExtractor(**motion_config)
            self._frame_buffer = {}  # 每个视频序列的帧缓冲
        else:
            self.motion_extractor = None

    def load_and_preprocess_image(self, file_path, **kwargs):
        """加载并预处理 RGB + Motion 图像"""
        # 1. 加载 RGB 图像
        im_rgb = imread(file_path)  # BGR, HxWx3

        if self.motion_mode == 'extract':
            # 2. 实时提取运动图
            # 假设同目录下有视频序列
            video_dir = str(Path(file_path).parent.parent.parent)
            seq_name = Path(file_path).stem.split('_')[0]  # 提取序列名

            # 获取帧索引
            frame_idx = self._get_frame_index(file_path)

            # 更新帧缓冲
            self._update_frame_buffer(seq_name, im_rgb, frame_idx)

            # 提取运动图
            motion_map = self.motion_extractor.extract(im_rgb)

        elif self.motion_mode == 'load':
            # 2. 加载预计算的运动图
            motion_path = self._get_motion_path(file_path)
            motion_map = imread(motion_path, cv2.IMREAD_GRAYSCALE)

            if len(motion_map.shape) == 2:
                motion_map = cv2.cvtColor(motion_map, cv2.COLOR_GRAY2BGR)
        else:
            raise ValueError(f"Unknown motion_mode: {self.motion_mode}")

        # 3. 合并 RGB + Motion
        im = self._merge_rgb_motion(im_rgb, motion_map)

        return im

    def _merge_rgb_motion(self, rgb, motion):
        """合并 RGB 和 Motion 通道"""
        h_rgb, w_rgb = rgb.shape[:2]
        h_m, w_m = motion.shape[:2]

        # 确保尺寸一致
        if h_rgb != h_m or w_rgb != w_m:
            motion = cv2.resize(motion, (w_rgb, h_rgb))

        # 早期融合: 6通道
        if motion.shape[2] == 3:  # 3通道运动图
            return np.concatenate([rgb, motion], axis=2)
        else:  # 单通道运动图
            return np.concatenate([rgb, cv2.merge([motion, motion, motion])], axis=2)

    def _update_frame_buffer(self, seq_name, frame, frame_idx):
        """更新帧缓冲"""
        if seq_name not in self._frame_buffer:
            self._frame_buffer[seq_name] = {
                'frames': [],
                'last_idx': -1
            }

        buf = self._frame_buffer[seq_name]

        # 检查帧是否连续
        if frame_idx != buf['last_idx'] + 1:
            # 帧不连续，重置缓冲
            buf['frames'] = []

        buf['frames'].append(frame)
        buf['last_idx'] = frame_idx

        # 更新提取器的帧缓冲
        if hasattr(self.motion_extractor, 'frame_buffer'):
            self.motion_extractor.frame_buffer.clear()
            self.motion_extractor.frame_buffer.extend(buf['frames'][-self.motion_extractor.num_frames:])

    def _get_frame_index(self, file_path):
        """从文件名提取帧索引"""
        stem = Path(file_path).stem
        # 假设命名格式: seqname_frameID.jpg
        try:
            return int(stem.split('_')[-1])
        except:
            return 0

    def _get_motion_path(self, rgb_path):
        """获取预计算运动图的路径"""
        rgb_path = Path(rgb_path)
        # visible/train/xxx.jpg -> motion/train/xxx.jpg
        motion_path = rgb_path.parent.parent / 'motion' / rgb_path.parent.name / rgb_path.name
        return str(motion_path)
```

## 三、MoE 模块适配

### 3.1 现有 MoE 模块通道适配

YOLO-Master 的 MoE 模块（如 `OptimizedMOEImproved`）使用 `nn.Conv2d` 作为路由器和专家，需要适配输入通道数:

```python
# ultralytics/nn/modules/moe/channel_adapter.py

class ChannelAdapterMoE(nn.Module):
    """
    通道适配 MoE 模块
    自动处理不同输入通道数的多模态融合
    """

    def __init__(self, in_channels, out_channels, num_experts=4, top_k=2,
                 rgb_channels=3, motion_channels=3, **moe_kwargs):
        super().__init__()

        self.in_channels = in_channels
        self.out_channels = out_channels
        self.rgb_channels = rgb_channels
        self.motion_channels = motion_channels

        # 基础 MoE 模块 (使用 in_channels)
        self.moe = OptimizedMOEImproved(
            in_channels=in_channels,
            out_channels=out_channels,
            num_experts=num_experts,
            top_k=top_k,
            **moe_kwargs
        )

        # 模态感知路由器 (可选增强)
        self.modality_router = nn.Sequential(
            nn.AdaptiveAvgPool2d(1),
            nn.Conv2d(in_channels, 2, 1),  # 2 = RGB/Motion 决策
            nn.Softmax(dim=1)
        )

    def forward(self, x):
        """
        Args:
            x: (B, in_channels, H, W) 输入张量

        Returns:
            输出张量 (B, out_channels, H, W)
        """
        # 标准 MoE 前向
        out = self.moe(x)

        return out
```

## 六、推理使用

```python
# predict_rgbmotion.py

from ultralytics import YOLO
import cv2
import numpy as np

def main():
    # 加载模型
    model = YOLO('runs/rgbmotion/yolo-master-n/weights/best.pt')

    # 方式1: 图片推理
    results = model.predict(
        source='path/to/image.jpg',
        imgsz=640,
        conf=0.25,
        device='0',
    )

    # 方式2: 视频流推理 (带运动信息)
    # 实时提取运动信息
    cap = cv2.VideoCapture('video.mp4')

    frames = []
    while True:
        ret, frame = cap.read()
        if not ret:
            break

        frames.append(frame)
        if len(frames) > 3:
            frames.pop(0)

        if len(frames) == 3:
            # 提取运动图
            motion = extract_motion_simple(frames)

            # 合并 RGB + Motion
            merged = np.concatenate([frame, motion], axis=2)

            # 推理
            results = model.predict(merged, verbose=False)

            # 可视化
            annotated = results[0].plot()
            cv2.imshow('Result', annotated)
            cv2.waitKey(1)

    cap.release()

def extract_motion_simple(frames):
    """简化版运动提取 (用于推理)"""
    # 使用相邻帧差分
    gray_frames = [cv2.cvtColor(f, cv2.COLOR_BGR2GRAY) for f in frames]

    # 计算帧差
    diff1 = cv2.absdiff(gray_frames[0], gray_frames[1])
    diff2 = cv2.absdiff(gray_frames[1], gray_frames[2])

    # 合并差异
    motion = cv2.addWeighted(diff1, 0.5, diff2, 0.5, 0)

    # 归一化
    if motion.max() > 0:
        motion = ((motion - motion.min()) / (motion.max() - motion.min()) * 255).astype(np.uint8)

    # 转3通道
    motion = cv2.merge([motion, motion, motion])

    return motion

if __name__ == '__main__':
    main()
```
